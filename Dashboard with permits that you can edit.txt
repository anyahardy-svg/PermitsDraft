import React, { useState } from 'react';
import {
  View,
  Text,
  TextInput,
  TouchableOpacity,
  ScrollView,
  StyleSheet,
  Alert,
  FlatList
} from 'react-native';

const PermitManagementApp = () => {
  const [currentScreen, setCurrentScreen] = useState('dashboard');
  const [selectedPermit, setSelectedPermit] = useState(null);
  const [isEditMode, setIsEditMode] = useState(false);
  
  const [permits, setPermits] = useState([
    {
      id: 'PTW-001',
      type: 'Hot Work',
      description: 'Welding repairs on conveyor frame',
      requestedBy: 'John Smith',
      location: 'Processing Plant A',
      status: 'pending_approval',
      priority: 'high',
      submittedDate: '2025-01-15',
      startDate: '2025-01-16',
      startTime: '08:00',
      endDate: '2025-01-16',
      endTime: '17:00'
    },
    {
      id: 'PTW-002', 
      type: 'Confined Space',
      description: 'Tank cleaning and inspection',
      requestedBy: 'Mike Johnson',
      location: 'Storage Area B',
      status: 'pending_inspection',
      priority: 'high',
      submittedDate: '2025-01-14'
    },
    {
      id: 'PTW-003',
      type: 'Electrical',
      description: 'Panel maintenance',
      requestedBy: 'Sarah Davis',
      location: 'Control Room',
      status: 'active',
      priority: 'medium',
      submittedDate: '2025-01-13'
    },
    {
      id: 'PTW-004',
      type: 'Working at Height',
      description: 'Roof inspection and repairs',
      requestedBy: 'Tom Wilson',
      location: 'Building C',
      status: 'completed',
      priority: 'medium',
      submittedDate: '2025-01-10'
    }
  ]);

  const [formData, setFormData] = useState({
    workDescription: '',
    location: '',
    requestedBy: '',
    department: '',
    startDate: '',
    endDate: '',
    startTime: '',
    endTime: '',
    priority: 'medium'
  });

  // Navigation
  const navigateToScreen = (screenName) => {
    console.log('Navigating to:', screenName);
    setCurrentScreen(screenName);
  };

  // Permit selection and editing
  const selectPermit = (permit) => {
    setSelectedPermit(permit);
    setFormData({
      workDescription: permit.description || '',
      location: permit.location || '',
      requestedBy: permit.requestedBy || '',
      department: permit.department || '',
      startDate: permit.startDate || '',
      endDate: permit.endDate || '',
      startTime: permit.startTime || '',
      endTime: permit.endTime || '',
      priority: permit.priority || 'medium'
    });
    setCurrentScreen('permit_detail');
  };

  const enableEditMode = () => {
    setIsEditMode(true);
  };

  const cancelEdit = () => {
    setIsEditMode(false);
  };

  const savePermitChanges = () => {
    const updatedPermits = permits.map(permit => 
      permit.id === selectedPermit.id 
        ? {
            ...permit,
            description: formData.workDescription,
            location: formData.location,
            requestedBy: formData.requestedBy,
            department: formData.department,
            startDate: formData.startDate,
            endDate: formData.endDate,
            startTime: formData.startTime,
            endTime: formData.endTime,
            priority: formData.priority,
            lastModified: new Date().toISOString().split('T')[0]
          }
        : permit
    );
    
    setPermits(updatedPermits);
    setSelectedPermit({
      ...selectedPermit,
      description: formData.workDescription,
      location: formData.location,
      requestedBy: formData.requestedBy,
      department: formData.department,
      startDate: formData.startDate,
      endDate: formData.endDate,
      startTime: formData.startTime,
      endTime: formData.endTime,
      priority: formData.priority,
      lastModified: new Date().toISOString().split('T')[0]
    });
    
    setIsEditMode(false);
    Alert.alert('Changes Saved', 'Permit has been updated successfully');
  };

  const deletePermit = () => {
    Alert.alert(
      'Delete Permit',
      `Are you sure you want to delete permit ${selectedPermit.id}?`,
      [
        { text: 'Cancel', style: 'cancel' },
        { 
          text: 'Delete', 
          style: 'destructive',
          onPress: () => {
            const updatedPermits = permits.filter(permit => permit.id !== selectedPermit.id);
            setPermits(updatedPermits);
            setSelectedPermit(null);
            navigateToScreen('dashboard');
          }
        }
      ]
    );
  };

  const submitPermit = () => {
    const newPermit = {
      id: `PTW-${String(permits.length + 1).padStart(3, '0')}`,
      type: 'General',
      description: formData.workDescription,
      requestedBy: formData.requestedBy,
      location: formData.location,
      status: 'pending_approval',
      priority: formData.priority,
      submittedDate: new Date().toISOString().split('T')[0],
      startDate: formData.startDate,
      startTime: formData.startTime,
      endDate: formData.endDate,
      endTime: formData.endTime,
      department: formData.department
    };

    setPermits([...permits, newPermit]);
    
    // Reset form
    setFormData({
      workDescription: '',
      location: '',
      requestedBy: '',
      department: '',
      startDate: '',
      endDate: '',
      startTime: '',
      endTime: '',
      priority: 'medium'
    });
    
    Alert.alert('Permit Submitted', `Permit ${newPermit.id} submitted for approval`, [
      { text: 'OK', onPress: () => navigateToScreen('dashboard') }
    ]);
  };
  const getDashboardCounts = () => {
    return {
      pendingApproval: permits.filter(p => p.status === 'pending_approval').length,
      pendingInspection: permits.filter(p => p.status === 'pending_inspection').length,
      active: permits.filter(p => p.status === 'active').length,
      completed: permits.filter(p => p.status === 'completed').length
    };
  };

  const getPriorityColor = (priority) => {
    switch(priority) {
      case 'high': return '#EF4444';
      case 'medium': return '#F59E0B';
      case 'low': return '#10B981';
      default: return '#6B7280';
    }
  };

  const getStatusColor = (status) => {
    switch(status) {
      case 'pending_approval': return '#F59E0B';
      case 'pending_inspection': return '#EF4444';
      case 'active': return '#10B981';
      case 'completed': return '#6B7280';
      default: return '#6B7280';
    }
  };

  const getStatusText = (status) => {
    switch(status) {
      case 'pending_approval': return 'Pending Approval';
      case 'pending_inspection': return 'Needs Inspection';
      case 'active': return 'Active';
      case 'completed': return 'Completed';
      default: return status;
    }
  };

  // Render functions
  const renderPermitItem = ({ item: permit }) => (
    <TouchableOpacity 
      style={styles.permitListCard}
      onPress={() => selectPermit(permit)}
    >
      <View style={styles.permitListHeader}>
        <Text style={styles.permitId}>{permit.id}</Text>
        <View style={[styles.statusBadge, { backgroundColor: getStatusColor(permit.status) }]}>
          <Text style={styles.statusText}>{getStatusText(permit.status)}</Text>
        </View>
      </View>
      
      <Text style={styles.permitType}>{permit.type}</Text>
      <Text style={styles.permitDescription}>{permit.description}</Text>
      
      <View style={styles.permitDetails}>
        <Text style={styles.detailText}>Location: {permit.location}</Text>
        <Text style={styles.detailText}>Requested by: {permit.requestedBy}</Text>
        <Text style={styles.detailText}>Date: {permit.submittedDate}</Text>
        {permit.lastModified && (
          <Text style={styles.detailText}>Last modified: {permit.lastModified}</Text>
        )}
      </View>
      
      <View style={[styles.priorityIndicator, { backgroundColor: getPriorityColor(permit.priority) }]}>
        <Text style={styles.priorityText}>{permit.priority.toUpperCase()}</Text>
      </View>
      
      <View style={styles.tapToViewIndicator}>
        <Text style={styles.tapToViewText}>Tap to view details</Text>
      </View>
    </TouchableOpacity>
  );

  const renderDashboard = () => {
    const counts = getDashboardCounts();
    
    return (
      <View style={styles.screenContainer}>
        <View style={styles.header}>
          <Text style={styles.title}>Permit Management Dashboard</Text>
        </View>

        <ScrollView style={styles.content}>
          <View style={styles.dashboardGrid}>
            <TouchableOpacity 
              style={[styles.dashboardCard, { borderLeftColor: '#F59E0B' }]}
              onPress={() => navigateToScreen('pending_approval')}
            >
              <Text style={styles.cardNumber}>{counts.pendingApproval}</Text>
              <Text style={styles.cardLabel}>Pending Approval</Text>
            </TouchableOpacity>

            <TouchableOpacity 
              style={[styles.dashboardCard, { borderLeftColor: '#EF4444' }]}
              onPress={() => navigateToScreen('pending_inspection')}
            >
              <Text style={styles.cardNumber}>{counts.pendingInspection}</Text>
              <Text style={styles.cardLabel}>Needs Inspection</Text>
            </TouchableOpacity>

            <TouchableOpacity 
              style={[styles.dashboardCard, { borderLeftColor: '#10B981' }]}
              onPress={() => navigateToScreen('active')}
            >
              <Text style={styles.cardNumber}>{counts.active}</Text>
              <Text style={styles.cardLabel}>Active Permits</Text>
            </TouchableOpacity>

            <TouchableOpacity 
              style={[styles.dashboardCard, { borderLeftColor: '#6B7280' }]}
              onPress={() => navigateToScreen('completed')}
            >
              <Text style={styles.cardNumber}>{counts.completed}</Text>
              <Text style={styles.cardLabel}>Completed</Text>
            </TouchableOpacity>
          </View>

          <TouchableOpacity 
            style={styles.primaryButton}
            onPress={() => navigateToScreen('new_permit')}
          >
            <Text style={styles.primaryButtonText}>Create New Permit</Text>
          </TouchableOpacity>
        </ScrollView>
      </View>
    );
  };

  const renderPermitList = (status, title) => {
    const filteredPermits = permits.filter(p => p.status === status);
    
    return (
      <View style={styles.screenContainer}>
        <View style={styles.header}>
          <TouchableOpacity onPress={() => navigateToScreen('dashboard')}>
            <Text style={styles.backButton}>‚Üê Back</Text>
          </TouchableOpacity>
          <Text style={styles.title}>{title}</Text>
        </View>

        <View style={styles.content}>
          {filteredPermits.length === 0 ? (
            <View style={styles.emptyState}>
              <Text style={styles.emptyStateText}>No {title.toLowerCase()} found</Text>
            </View>
          ) : (
            <FlatList
              data={filteredPermits}
              renderItem={renderPermitItem}
              keyExtractor={(item) => item.id}
              style={styles.permitList}
              contentContainerStyle={styles.permitListContainer}
            />
          )}
        </View>
      </View>
    );
  };

  const renderPermitDetail = () => {
    if (!selectedPermit) return null;

    return (
      <View style={styles.screenContainer}>
        <View style={styles.header}>
          <TouchableOpacity onPress={() => {
            setSelectedPermit(null);
            setIsEditMode(false);
            navigateToScreen('dashboard');
          }}>
            <Text style={styles.backButton}>‚Üê Back</Text>
          </TouchableOpacity>
          <Text style={styles.title}>Permit {selectedPermit.id}</Text>
          {!isEditMode && (
            <TouchableOpacity onPress={enableEditMode}>
              <Text style={styles.editButton}>Edit</Text>
            </TouchableOpacity>
          )}
        </View>

        <ScrollView style={styles.content}>
          {isEditMode && (
            <View style={styles.editModeHeader}>
              <Text style={styles.editModeTitle}>Editing Permit {selectedPermit.id}</Text>
              <View style={styles.editModeButtons}>
                <TouchableOpacity style={styles.cancelButton} onPress={cancelEdit}>
                  <Text style={styles.cancelButtonText}>Cancel</Text>
                </TouchableOpacity>
                <TouchableOpacity style={styles.saveButton} onPress={savePermitChanges}>
                  <Text style={styles.saveButtonText}>Save Changes</Text>
                </TouchableOpacity>
              </View>
            </View>
          )}

          {!isEditMode && (
            <View style={styles.permitDetailHeader}>
              <View style={styles.permitDetailInfo}>
                <Text style={styles.permitDetailTitle}>{selectedPermit.type} Permit</Text>
                <View style={[styles.statusBadge, { backgroundColor: getStatusColor(selectedPermit.status) }]}>
                  <Text style={styles.statusText}>{getStatusText(selectedPermit.status)}</Text>
                </View>
              </View>
              <View style={styles.permitDetailActions}>
                <TouchableOpacity style={styles.actionButton} onPress={enableEditMode}>
                  <Text style={styles.actionButtonText}>‚úèÔ∏è Edit</Text>
                </TouchableOpacity>
                <TouchableOpacity style={[styles.actionButton, styles.deleteButton]} onPress={deletePermit}>
                  <Text style={[styles.actionButtonText, styles.deleteButtonText]}>üóëÔ∏è Delete</Text>
                </TouchableOpacity>
              </View>
            </View>
          )}

          <View style={styles.section}>
            <View style={styles.sectionHeader}>
              <Text style={styles.sectionTitle}>General Information</Text>
            </View>
            
            <View style={styles.sectionContent}>
              {isEditMode ? (
                <>
                  <Text style={styles.label}>Work Description</Text>
                  <TextInput
                    style={[styles.input, styles.textArea]}
                    multiline
                    numberOfLines={3}
                    value={formData.workDescription}
                    onChangeText={(text) => setFormData({...formData, workDescription: text})}
                    placeholder="Describe the work to be performed..."
                  />
                  
                  <Text style={styles.label}>Location</Text>
                  <TextInput
                    style={styles.input}
                    value={formData.location}
                    onChangeText={(text) => setFormData({...formData, location: text})}
                    placeholder="Work location"
                  />
                  
                  <Text style={styles.label}>Requested By</Text>
                  <TextInput
                    style={styles.input}
                    value={formData.requestedBy}
                    onChangeText={(text) => setFormData({...formData, requestedBy: text})}
                    placeholder="Your name"
                  />

                  <Text style={styles.label}>Department</Text>
                  <TextInput
                    style={styles.input}
                    value={formData.department}
                    onChangeText={(text) => setFormData({...formData, department: text})}
                    placeholder="Department/Team"
                  />

                  <View style={styles.dateTimeRow}>
                    <View style={styles.dateTimeColumn}>
                      <Text style={styles.label}>Start Date</Text>
                      <TextInput
                        style={styles.input}
                        value={formData.startDate}
                        onChangeText={(text) => setFormData({...formData, startDate: text})}
                        placeholder="YYYY-MM-DD"
                      />
                    </View>

                    <View style={styles.dateTimeColumn}>
                      <Text style={styles.label}>Start Time</Text>
                      <TextInput
                        style={styles.input}
                        value={formData.startTime}
                        onChangeText={(text) => setFormData({...formData, startTime: text})}
                        placeholder="HH:MM"
                      />
                    </View>
                  </View>

                  <View style={styles.dateTimeRow}>
                    <View style={styles.dateTimeColumn}>
                      <Text style={styles.label}>End Date</Text>
                      <TextInput
                        style={styles.input}
                        value={formData.endDate}
                        onChangeText={(text) => setFormData({...formData, endDate: text})}
                        placeholder="YYYY-MM-DD"
                      />
                    </View>

                    <View style={styles.dateTimeColumn}>
                      <Text style={styles.label}>End Time</Text>
                      <TextInput
                        style={styles.input}
                        value={formData.endTime}
                        onChangeText={(text) => setFormData({...formData, endTime: text})}
                        placeholder="HH:MM"
                      />
                    </View>
                  </View>

                  <Text style={styles.label}>Priority</Text>
                  <View style={styles.priorityButtons}>
                    {['low', 'medium', 'high'].map(priority => (
                      <TouchableOpacity
                        key={priority}
                        style={[
                          styles.priorityButton,
                          { backgroundColor: formData.priority === priority ? getPriorityColor(priority) : '#E5E7EB' }
                        ]}
                        onPress={() => {
                          const updatedFormData = {...formData};
                          updatedFormData.priority = priority;
                          setFormData(updatedFormData);
                        }}
                      >
                        <Text style={[
                          styles.priorityButtonText,
                          { color: formData.priority === priority ? 'white' : '#374151' }
                        ]}>
                          {priority.toUpperCase()}
                        </Text>
                      </TouchableOpacity>
                    ))}
                  </View>
                </>
              ) : (
                <>
                  <View style={styles.detailItem}>
                    <Text style={styles.detailLabel}>Work Description:</Text>
                    <Text style={styles.detailValue}>{selectedPermit.description || 'Not specified'}</Text>
                  </View>
                  
                  <View style={styles.detailItem}>
                    <Text style={styles.detailLabel}>Location:</Text>
                    <Text style={styles.detailValue}>{selectedPermit.location || 'Not specified'}</Text>
                  </View>
                  
                  <View style={styles.detailItem}>
                    <Text style={styles.detailLabel}>Requested By:</Text>
                    <Text style={styles.detailValue}>{selectedPermit.requestedBy || 'Not specified'}</Text>
                  </View>

                  <View style={styles.detailRow}>
                    <View style={styles.detailColumn}>
                      <Text style={styles.detailLabel}>Start Date:</Text>
                      <Text style={styles.detailValue}>{selectedPermit.startDate || 'Not set'}</Text>
                    </View>
                    <View style={styles.detailColumn}>
                      <Text style={styles.detailLabel}>Start Time:</Text>
                      <Text style={styles.detailValue}>{selectedPermit.startTime || 'Not set'}</Text>
                    </View>
                  </View>

                  <View style={styles.detailRow}>
                    <View style={styles.detailColumn}>
                      <Text style={styles.detailLabel}>End Date:</Text>
                      <Text style={styles.detailValue}>{selectedPermit.endDate || 'Not set'}</Text>
                    </View>
                    <View style={styles.detailColumn}>
                      <Text style={styles.detailLabel}>End Time:</Text>
                      <Text style={styles.detailValue}>{selectedPermit.endTime || 'Not set'}</Text>
                    </View>
                  </View>

                  <View style={styles.detailItem}>
                    <Text style={styles.detailLabel}>Priority:</Text>
                    <View style={[styles.priorityBadge, { backgroundColor: getPriorityColor(selectedPermit.priority) }]}>
                      <Text style={styles.priorityBadgeText}>{selectedPermit.priority?.toUpperCase()}</Text>
                    </View>
                  </View>

                  <View style={styles.detailItem}>
                    <Text style={styles.detailLabel}>Submitted:</Text>
                    <Text style={styles.detailValue}>{selectedPermit.submittedDate}</Text>
                  </View>

                  {selectedPermit.lastModified && (
                    <View style={styles.detailItem}>
                      <Text style={styles.detailLabel}>Last Modified:</Text>
                      <Text style={styles.detailValue}>{selectedPermit.lastModified}</Text>
                    </View>
                  )}
                </>
              )}
            </View>
          </View>
        </ScrollView>
      </View>
    );
  };

  const renderNewPermit = () => {
    return (
      <View style={styles.screenContainer}>
        <View style={styles.header}>
          <TouchableOpacity onPress={() => navigateToScreen('dashboard')}>
            <Text style={styles.backButton}>‚Üê Back</Text>
          </TouchableOpacity>
          <Text style={styles.title}>New Permit Request</Text>
        </View>
        
        <View style={styles.content}>
          <View style={{padding: 20}}>
            <Text style={{fontSize: 18, marginBottom: 20}}>Create New Permit - Test Screen</Text>
            <TouchableOpacity 
              style={styles.primaryButton}
              onPress={() => navigateToScreen('dashboard')}
            >
              <Text style={styles.primaryButtonText}>Back to Dashboard</Text>
            </TouchableOpacity>
          </View>
        </View>
      </View>
    );
  };
  console.log('Current screen:', currentScreen); // Debug line
  
  if (currentScreen === 'permit_detail') {
    return renderPermitDetail();
  }
  
  if (currentScreen === 'pending_approval') {
    return renderPermitList('pending_approval', 'Pending Approval');
  }
  
  if (currentScreen === 'pending_inspection') {
    return renderPermitList('pending_inspection', 'Needs Inspection');
  }
  
  if (currentScreen === 'active') {
    return renderPermitList('active', 'Active Permits');
  }
  
  if (currentScreen === 'completed') {
    return renderPermitList('completed', 'Completed Permits');
  }

  return renderDashboard();
};
